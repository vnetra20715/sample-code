using OpenAI.Chat;
using Azure.AI.OpenAI;
using Azure;
using Microsoft.Extensions.Configuration;

 public abstract class AgentBase : BusinessBase
 {
     private AzureOpenAIClient _client = null;
     private ChatClient _chatClient = null;
     private string _relatedQuestionSystemPrompt = "";

     public AgentBase(IConfiguration configuration, IMapper mapper, AppDbContext dbContext, QueryHelper queryHelper, IBusinessHelper businessHelper)
         : base(configuration, mapper, dbContext, queryHelper, businessHelper)
     { }

     protected async Task<string> GetAiResponse(string systemPrompt, List<string> otherPrompts)
     {
         List<ChatMessage> messages = new List<ChatMessage>();
         SetupClient();
         var options = new ChatCompletionOptions
         {
             Temperature = 0.0f,
             //MaxOutputTokenCount = 4000,
             FrequencyPenalty = 0.0f,
             PresencePenalty = 0.0f,
         };
         messages.Add(ChatMessage.CreateSystemMessage(systemPrompt));
         foreach (var prompt in otherPrompts)
         {
             messages.Add(ChatMessage.CreateUserMessage(prompt));
         }
         var response = await _chatClient.CompleteChatAsync(messages, options);
         return response.Value.Content[0].Text;
     }
     
     protected async Task <List<string>> GetRelatedQuestions(List<string> messages)
     {
         List<string> relatedQuestions = new List<string>();
         if (string.IsNullOrEmpty(_relatedQuestionSystemPrompt))
         {
             string path = Path.Combine(AppDomain.CurrentDomain.BaseDirectory, "Prompts", "related_questions_v1.txt");
             _relatedQuestionSystemPrompt = File.ReadAllText(path);
         }
         var json = await GetAiResponse(_relatedQuestionSystemPrompt, messages);
         json = GetClearJson(json);
         relatedQuestions = json.GetFromJson<List<string>>();
         return relatedQuestions;
     }

     protected AgentType GetAgentType(string agentName)
     {
         if (string.IsNullOrWhiteSpace(agentName))
             throw new ArgumentException("Agent name cannot be null or empty.");

         switch (agentName.Trim())
         {
            case "General Query Agent":
                 return AgentType.GeneralQueryAgent;

             default:
                 throw new ArgumentException($"Unknown agent type: {agentName}");
         }
     }

     private void SetupClient()
     {
         if (_client == null)
         {
             _client = new AzureOpenAIClient(new Uri(_configuration["OpenAI:Url"]), new AzureKeyCredential(_configuration["OpenAI:ApiKey"]));
         }
         if (_chatClient == null)
         {
             _chatClient = _client.GetChatClient(_configuration["OpenAI:DeploymentName"]);
         }
     }
 }

 public class GeneralQueryAgent : AgentBase, IChatAgent
 {
     private readonly string agentName = "General Query Agent";
     private string _systemPrompt = "";
     public GeneralQueryAgent(IConfiguration configuration) : base(configuration, null, null, null, null)
     {

     }

     public string AgentName => agentName;

     public async Task<AgentResultDto> GetChatResponseAsync(ChatRequestDto request, List<string> messages)
     {
         if (string.IsNullOrEmpty(_systemPrompt))
         {
             string path = Path.Combine(AppDomain.CurrentDomain.BaseDirectory, "Prompts", "general_query_agent_v1.txt");
             _systemPrompt = File.ReadAllText(path);
         }
         if (request.IsLast)
         {
             var task1 = GetAiResponse(_systemPrompt, messages);
             var task2 = GetRelatedQuestions(messages);
             await Task.WhenAll(task1, task2);
             var output = new AgentResultDto(agentName, task1.Result);
             output.RelatedQuestions = task2.Result;
             return output;
         }
         else
         {
             var result = await GetAiResponse(_systemPrompt, messages);
             return new AgentResultDto(agentName, result);
         }
     }
 }

 public class AppDbContext : DbContext
{
    private readonly IConfiguration _configuration;
    private readonly IAppLogger _logger;

    public AppDbContext(IConfiguration configuration, IAppLogger logger)
    {
        _configuration = configuration;
        _logger = logger;
    }
    protected override void OnConfiguring(DbContextOptionsBuilder optionsBuilder)
    {
        string conStr = "";
        if (_configuration["ConnectionStrings:DbType"].GetString() == "MySql")
        {
            conStr = _configuration["ConnectionStrings:MySql"];
            optionsBuilder.UseMySql(conStr, ServerVersion.AutoDetect(conStr));
        }
        else
        {
            conStr = _configuration["ConnectionStrings:Sql"];
            optionsBuilder.UseSqlServer(conStr);
        }
        if (_configuration["LogMode"] == "Debug")
        {
            optionsBuilder.LogTo(_logger.Debug, LogLevel.Information);
        }
    }
   }

   public class QueryHelper
{
    private IConfiguration _configuration = null;
    private bool _isTransaction = false;
    private DbConnection _transactionConnection = null;
    private DbTransaction _transaction = null;

    public QueryHelper(IConfiguration configuration)
    {
        _configuration = configuration;
    }

    public async Task BeginTransaction()
    {
        var conn = GetDbConnection();
        await conn.OpenAsync();
        _isTransaction = true;
        _transactionConnection = conn;
        _transaction = await conn.BeginTransactionAsync();
    }

    public async Task CommitTransaction()
    {
        if (_isTransaction)
        {
            await _transaction.CommitAsync();
            _transactionConnection.Close();
            _isTransaction = false;
        }
    }

    public async Task RollbackTransaction()
    {
        if (_isTransaction)
        {
            await _transaction.RollbackAsync();
            _transactionConnection.Close();
            _isTransaction = false;
        }
    }

    private DbConnection GetDbConnection()
    {
        if (_configuration["ConnectionStrings:DbType"].GetString() == "MySql")
        {
            return new MySqlConnection(_configuration["ConnectionStrings:MySql"]);
        }
        else
        {
            return new SqlConnection(_configuration["ConnectionStrings:Sql"]);
        }
    }


    public async Task<IEnumerable<T>> Query<T>(string query, object param = null)
    {
        if (_transactionConnection != null)
        {
            return await SqlMapper.QueryAsync<T>(_transactionConnection, query, param, _transaction);
        }
        using (var conn = GetDbConnection())
        {
            return await SqlMapper.QueryAsync<T>(conn, query, param);
        }
    }

    public async Task ExecuteQuery(string query, object param = null)
    {
        if (_transactionConnection != null)
        {
            await SqlMapper.ExecuteAsync(_transactionConnection, query, param, _transaction);
            return;
        }
        using (var conn = GetDbConnection())
        {
            await SqlMapper.ExecuteAsync(conn, query, param);
        }
    }

    public async Task<T> ExecuteScalar<T>(string query, object param = null)
    {
        if (_transactionConnection != null)
        {
            var t = await _transactionConnection.ExecuteScalarAsync<T>(query, param, _transaction);
            return t;
        }
        using (var conn = GetDbConnection())
        {
            var t = await conn.ExecuteScalarAsync<T>(query, param);
            return t;
        }
    }

    public long? GetSingleId(string query, object param = null)
    {
        if (_transactionConnection == null)
        {
            return _transactionConnection.QuerySingleOrDefault<long?>(query, param, _transaction);
        }
        using (var conn = GetDbConnection())
        {
            return conn.QuerySingleOrDefault<long?>(query, param, _transaction);
        }
    }
}

     public List<Tuple<string, List<string>>> GetExcelHeaders(byte[] fileBytes)
     {

         var result = new List<Tuple<string, List<string>>>();

         using var stream = new MemoryStream(fileBytes);
         using var workbook = new XLWorkbook(stream);

         foreach (var worksheet in workbook.Worksheets)
         {
             var sheetName = FormatSheetName(worksheet.Name);
             var headers = new List<string>();

             var firstRow = worksheet.RangeUsed()?.RowsUsed().FirstOrDefault();
             if (firstRow != null)
             {
                 foreach (var cell in firstRow.Cells())
                 {
                     var value = cell.GetValue<string>()?.Trim();
                     if (!string.IsNullOrEmpty(value))
                     {
                         headers.Add(value);
                     }
                 }
             }
             if (headers.Count > 2)
             {
                 result.Add(new Tuple<string, List<string>>(sheetName, headers));
             }
         }

         return result;
     }

     private string FormatSheetName(string sheetName)
     {
         return sheetName;
     }

     public List<Tuple<string, List<ChargeDto>>> GetChargesByHeaders(byte[] fileBytes, List<ExcelSheetHeaderDto> sheetConfigs)
     {
         var result = new List<Tuple<string, List<ChargeDto>>>();
         using var stream = new MemoryStream(fileBytes);
         using var workbook = new XLWorkbook(stream);
         foreach (var config in sheetConfigs)
         {
             var worksheet = workbook.Worksheets.FirstOrDefault(x => x.Name.Equals(config.SheetName, StringComparison.OrdinalIgnoreCase));
             if (worksheet == null) continue;
             var sheetCharges = new List<ChargeDto>();
             var headerRow = worksheet.RangeUsed()?.RowsUsed().FirstOrDefault();
             if (headerRow == null) continue;
             var headerMap = new Dictionary<string, int>(StringComparer.OrdinalIgnoreCase);
             int colIndex = 1;
             foreach (var cell in headerRow.Cells())
             {
                 var val = cell.GetString().Trim();
                 if (!string.IsNullOrEmpty(val) && !headerMap.ContainsKey(val)) headerMap[val] = colIndex;
                 colIndex++;
             }
             var mappedIndex = new Dictionary<string, int>();
             foreach (var h in config.Headers)
             {
                 if (!string.IsNullOrEmpty(h.MatchedField) && headerMap.ContainsKey(h.MatchedField))
                     mappedIndex[h.TableFiled] = headerMap[h.MatchedField];
             }
             var rows = worksheet.RangeUsed()?.RowsUsed().Skip(1);
             if (rows == null) continue;
             foreach (var row in rows)
             {
                 if (row.Cell(1).IsEmpty())
                     continue;
                 if (row.Cell(1).GetString().Contains("TOTAL", StringComparison.OrdinalIgnoreCase)
                         || row.Cell(1).GetString().Contains("Summary", StringComparison.OrdinalIgnoreCase))
                     continue;

                 var dto = new ChargeDto { SheetName = config.SheetName };

                 if (mappedIndex.ContainsKey("PatientName"))
                     dto.PatientName = row.Cell(mappedIndex["PatientName"]).GetString().Trim();

                 if (mappedIndex.ContainsKey("FirstName"))
                     dto.FirstName = row.Cell(mappedIndex["FirstName"]).GetString().Trim();

                 if (mappedIndex.ContainsKey("LastName"))
                     dto.LastName = row.Cell(mappedIndex["LastName"]).GetString().Trim();

                 if (mappedIndex.ContainsKey("Insurance"))
                     dto.Insurance = row.Cell(mappedIndex["Insurance"]).GetString().Trim();

                 if (mappedIndex.ContainsKey("Facility"))
                     dto.Facility = row.Cell(mappedIndex["Facility"]).GetString().Trim();

                 if (mappedIndex.ContainsKey("PostDate"))
                     dto.PostDate = GetDate(row.Cell(mappedIndex["PostDate"]));

                 if (mappedIndex.ContainsKey("DateOfService"))
                     dto.DateOfService = GetDate(row.Cell(mappedIndex["DateOfService"]));

                 if (mappedIndex.ContainsKey("EncounterId"))
                     dto.EncounterId = row.Cell(mappedIndex["EncounterId"]).GetString().Trim();

                 if (mappedIndex.ContainsKey("PrimaryDiagnosis"))
                     dto.PrimaryDiagnosis = row.Cell(mappedIndex["PrimaryDiagnosis"]).GetString().Trim();

                 if (mappedIndex.ContainsKey("SecondaryDiagnosis"))
                     dto.SecondaryDiagnosis = row.Cell(mappedIndex["SecondaryDiagnosis"]).GetString().Trim();

                 if (mappedIndex.ContainsKey("CptCodeWithModifiers"))
                     dto.CptCodeWithModifiers = row.Cell(mappedIndex["CptCodeWithModifiers"]).GetString().Trim();

                 if (mappedIndex.ContainsKey("CptCode"))
                     dto.CptCode = row.Cell(mappedIndex["CptCode"]).GetString().Trim();

                 if (mappedIndex.ContainsKey("PrimaryInsurance"))
                     dto.PrimaryInsurance = row.Cell(mappedIndex["PrimaryInsurance"]).GetString().Trim();

                 if (mappedIndex.ContainsKey("SecondaryInsurance"))
                     dto.SecondaryInsurance = row.Cell(mappedIndex["PrimaryInsurance"]).GetString().Trim();

                 if (mappedIndex.ContainsKey("ProviderName"))
                     dto.ProviderName = row.Cell(mappedIndex["ProviderName"]).GetString().Trim();

                 if (mappedIndex.ContainsKey("TotalAmount"))
                 {
                     var cell = row.Cell(mappedIndex["TotalAmount"]);
                     dto.TotalAmount = GetDecimal(cell);
                 }
                 sheetCharges.Add(dto);
             }

             result.Add(new Tuple<string, List<ChargeDto>>(config.SheetName, sheetCharges));
         }
         return result;
     }

     private decimal? GetDecimal(IXLCell cell)
     {
         if (!cell.IsEmpty())
         {
             decimal amount;

             // Case 1: Numeric cell (covers int + decimal directly)
             if (cell.DataType == XLDataType.Number)
             {
                 return Convert.ToDecimal(cell.GetDouble());
             }
             else
             {
                 // Case 2: String (remove $, commas, spaces)
                 var raw = cell.GetString()
                               .Replace("#", "")
                               .Replace("$", "")
                               .Replace(",", "")
                               .Trim();

                 if (decimal.TryParse(raw, out amount))
                 {
                     return amount;
                 }
             }
         }
         return null;
     }

     private DateTime? GetDate(IXLCell cell)
     {
         if (cell.DataType == XLDataType.DateTime)
         {
             var date = cell.GetDateTime();
             return date.Date;
         }
         var value = cell.Value.GetString();
         string pattern = @"^(0[1-9]|1[0-2])/(0[1-9]|[12]\d|3[01])/(\d{4}|\d{2})$";
         if (Regex.IsMatch(value, pattern))
         {
             var culture = (CultureInfo)CultureInfo.InvariantCulture.Clone();
             culture.Calendar.TwoDigitYearMax = 2099; // Forces 00–99 → 2000–2099

             if (DateTime.TryParseExact(value,
                                        new[] { "MM/dd/yyyy", "MM/dd/yy" },
                                        culture,
                                        DateTimeStyles.None,
                                        out DateTime date))
             {
                 return date.Date;
             }
         }
         if (DateTime.TryParse(cell.GetString(), out var parsed))
             return parsed;

         return null;
     }
