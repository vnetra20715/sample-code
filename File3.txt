public async Task<List<ExcelSheetHeaderDto>> GetBillingHeaderDtos(BillingFileDto fileDto)
{
    var result = new List<ExcelSheetHeaderDto>();
    var semaphore = new SemaphoreSlim(5);
    BillingExcelReader billingExcelReader = new BillingExcelReader();
    string path = (fileDto.Type == BillingFileType.Charge) ?
                    Path.Combine(AppDomain.CurrentDomain.BaseDirectory, "Prompts", "charge-header-matching.txt")
                    : Path.Combine(AppDomain.CurrentDomain.BaseDirectory, "Prompts", "payment-header-matching.txt");
    var systemPrompt = File.ReadAllText(path);
    var headerList = billingExcelReader.GetExcelHeaders(fileDto.FileData);
    var tasks = new List<Task<List<BillingHeaderDto>>>();
    foreach (var header in headerList)
    {
        await semaphore.WaitAsync();
        tasks.Add(ProcessHeaderAsync(header, systemPrompt, semaphore));
    }
    await Task.WhenAll(tasks);
    foreach (var task in tasks)
    {
        var index = tasks.IndexOf(task);
        var obj = new ExcelSheetHeaderDto()
        {
            Headers = task.Result,
            SheetName = headerList[index].Item1
        };
        result.Add(obj);
    }
    return result;
}

      public List<Tuple<string, List<string>>> GetExcelHeaders(byte[] fileBytes)
      {

          var result = new List<Tuple<string, List<string>>>();

          using var stream = new MemoryStream(fileBytes);
          using var workbook = new XLWorkbook(stream);

          foreach (var worksheet in workbook.Worksheets)
          {
              var sheetName = FormatSheetName(worksheet.Name);
              var headers = new List<string>();

              var firstRow = worksheet.RangeUsed()?.RowsUsed().FirstOrDefault();
              if (firstRow != null)
              {
                  foreach (var cell in firstRow.Cells())
                  {
                      var value = cell.GetValue<string>()?.Trim();
                      if (!string.IsNullOrEmpty(value))
                      {
                          headers.Add(value);
                      }
                  }
              }
              if (headers.Count > 2)
              {
                  result.Add(new Tuple<string, List<string>>(sheetName, headers));
              }
          }

          return result;
      }

      private decimal? GetDecimal(IXLCell cell)
      {
          if (!cell.IsEmpty())
          {
              decimal amount;

              // Case 1: Numeric cell (covers int + decimal directly)
              if (cell.DataType == XLDataType.Number)
              {
                  return Convert.ToDecimal(cell.GetDouble());
              }
              else
              {
                  // Case 2: String (remove $, commas, spaces)
                  var raw = cell.GetString()
                                .Replace("#", "")
                                .Replace("$", "")
                                .Replace(",", "")
                                .Trim();

                  if (decimal.TryParse(raw, out amount))
                  {
                      return amount;
                  }
              }
          }
          return null;
      }

      private DateTime? GetDate(IXLCell cell)
      {
          if (cell.DataType == XLDataType.DateTime)
          {
              var date = cell.GetDateTime();
              return date.Date;
          }
          var value = cell.Value.GetString();
          string pattern = @"^(0[1-9]|1[0-2])/(0[1-9]|[12]\d|3[01])/(\d{4}|\d{2})$";
          if (Regex.IsMatch(value, pattern))
          {
              var culture = (CultureInfo)CultureInfo.InvariantCulture.Clone();
              culture.Calendar.TwoDigitYearMax = 2099; // Forces 00–99 → 2000–2099

              if (DateTime.TryParseExact(value,
                                         new[] { "MM/dd/yyyy", "MM/dd/yy" },
                                         culture,
                                         DateTimeStyles.None,
                                         out DateTime date))
              {
                  return date.Date;
              }
          }
          if (DateTime.TryParse(cell.GetString(), out var parsed))
              return parsed;

          return null;
      }

   private async Task<List<BillingHeaderDto>> ProcessHeaderAsync(Tuple<string, List<string>> header, string systemPrompt, SemaphoreSlim semaphore)
 {
     var tmp = new List<BillingHeaderDto>();
     try
     {
         var headersStr = $"Match following headers:\n{string.Join("\n", header.Item2)}";
         var aiJson = await _aiHelper.GetAiResponse(systemPrompt, new List<string> { headersStr });
         _appLogger.Info($"AI response for header matching of sheet {header.Item1}: {aiJson}");
         tmp = aiJson.GetFromJson<List<BillingHeaderDto>>();
     }
     finally
     {
         semaphore.Release();
     }
     return tmp;
 }
