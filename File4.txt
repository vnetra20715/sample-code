static void AddSession(WebApplicationBuilder builder)
{
    builder.Services.AddSession(options =>
    {
        // Set a short timeout for easy testing.
        options.IdleTimeout = TimeSpan.FromSeconds(60 * 60);
        options.Cookie.HttpOnly = true;
        options.Cookie.SecurePolicy = CookieSecurePolicy.SameAsRequest;
        //options.Cookie.SameSite = SameSiteMode.None;
        //options.IOTimeout = TimeSpan.FromSeconds(60 * 30);
    });
}

static void AddMapper(WebApplicationBuilder builder)
{
    var mapperConfig = new MapperConfiguration(mc =>
    {
        mc.AddProfile(new MappingProfile());
    });
    IMapper mapper = mapperConfig.CreateMapper();
    builder.Services.AddSingleton(mapper);
}

static void AddAuthentication(WebApplicationBuilder builder)
{
    builder.Services.AddAuthentication(CookieAuthenticationDefaults.AuthenticationScheme)
              .AddCookie(options =>
              {
                  options.LogoutPath = "/login";
                  options.LoginPath = "/login";
                  options.ReturnUrlParameter = "return-url";
                  //options.Cookie.Expiration = TimeSpan.FromMinutes(15);
                  options.ExpireTimeSpan = TimeSpan.FromMinutes(60);
                  options.SlidingExpiration = true;
              });

    builder.Services.AddAuthorization(options =>
    {
        options.AddPolicy(AppConstant.InternalUserPolicy, policy =>
                                policy.RequireAssertion(context =>
                                    context.User.HasClaim(c => c.Type == AppConstant.InternalUserPolicy)));
        options.AddPolicy(AppConstant.AccountUserPolicy, policy =>
                                policy.RequireAssertion(context =>
                                    context.User.HasClaim(c => c.Type == AppConstant.AccountUserPolicy)));
    });
}


static IConfiguration GetIConfiguration()
{
    var evn = System.Environment.GetEnvironmentVariable("ASPNETCORE_ENVIRONMENT");
    if (string.IsNullOrEmpty(evn) == false)
    {
        evn = "." + evn;
    }
    IConfiguration configuration = new ConfigurationBuilder()
                                    //.SetBasePath(Directory.GetCurrentDirectory())
                                    //.AddJsonFile($"appsettings{evn}.json")
                                    .AddJsonFile("appsettings.json", optional: true, reloadOnChange: true)
                                    .AddJsonFile($"appsettings{evn}.json", true, true)
                                    .AddEnvironmentVariables()
                                    .Build();
    if (configuration["ConnectionStrings:DbType"] == "MySql")
    {
        AppConstant.MySql = true;
    }
    return configuration;
}
