        public static bool IsLast<T>(this IList<T> list, T item)
        {
            if (list == null || list.Count == 0) return false;

            // Use EqualityComparer to handle nulls and different types correctly
            return EqualityComparer<T>.Default.Equals(list[list.Count - 1], item);
        }

        }

        public static string BreakIfNeeded(this string input, int chunkSize = 20)
        {
            if (string.IsNullOrEmpty(input))
                return input;

            // If there's already a space within chunkSize, skip processing
            if (Regex.IsMatch(input, $@"\S{{1,{chunkSize}}}\s"))
                return input;

            // Insert space every chunkSize characters
            return Regex.Replace(input, $@"(.{{{chunkSize}}})", "$1 ");
        }

        public static string GetPropertyDescription<T>(this T input, string propertyName)
        {
            PropertyDescriptor propDesc = TypeDescriptor.GetProperties(typeof(T))[propertyName];
            if (propDesc == null)
            {
                throw new ArgumentException($"Property '{propertyName}' not found on type '{typeof(T).FullName}'.");
            }

            DescriptionAttribute descriptionAttribute = (DescriptionAttribute)propDesc.Attributes[typeof(DescriptionAttribute)];

            return descriptionAttribute != null ? descriptionAttribute.Description : "No description available.";
        }

        public static string RemoveSpecialCharacters(this string input)
        {
            if (string.IsNullOrEmpty(input))
            {
                return "";
            }
            string pattern = "[^a-zA-Z0-9 ]";
            string replacement = ""; // Replace special characters with an underscore
            Regex regex = new Regex(pattern);
            return regex.Replace(input, replacement);
        }

        public static string ReplaceSpecialCharacters(this string input, string replace)
        {
            if (string.IsNullOrEmpty(input))
            {
                return string.Empty;
            }
            // Regex pattern: anything that's not a-z, A-Z, 0-9, or space
            string pattern = @"[^a-zA-Z0-9 ]";
            // Use the passed replacement instead of hardcoded value
            Regex regex = new Regex(pattern, RegexOptions.Compiled);
            return regex.Replace(input, replace);
        }

        public static List<List<T>> ChunkBy<T>(this List<T> source, int chunkSize)
        {
            return source
                .Select((x, i) => new { Index = i, Value = x })
                .GroupBy(x => x.Index / chunkSize)
                .Select(x => x.Select(v => v.Value).ToList())
                .ToList();
        }

        public static T TrimStringProperties<T>(this T input) where T : class
        {
            var stringProperties = input.GetType().GetProperties()
                .Where(p => p.PropertyType == typeof(string) && p.CanWrite);

            foreach (var stringProperty in stringProperties)
            {
                string currentValue = (string)stringProperty.GetValue(input, null);
                if (currentValue != null)
                {
                    stringProperty.SetValue(input, currentValue.Trim(), null);
                }
            }
            return input;
        }

        public static IDictionary<string, object> ToDictionary(this object obj)
        {
            IDictionary<string, object> result = new Dictionary<string, object>();

        public static string GetMD5(this string input)
        {
            using (var provider = System.Security.Cryptography.MD5.Create())
            {
                StringBuilder builder = new StringBuilder();

                foreach (byte b in provider.ComputeHash(Encoding.UTF8.GetBytes(input)))
                {
                    builder.Append(b.ToString("x2").ToLower());
                }

                return builder.ToString();
            }
        }

        public static string Description<T>(this T value) where T : struct
        {
            var enumType = value.GetType();
            var field = enumType.GetField(value.ToString());
            if (field != null)
            {
                var attributes = field.GetCustomAttributes(typeof(DescriptionAttribute), false);
                return attributes.Length == 0 ? value.ToString() : ((DescriptionAttribute)attributes[0]).Description;
            }
            return "";
        }

        private static string EncryptAes256(byte[] key, byte[] iv, string input)
        {
            if (key.Length * 8 != encDec.KeySize)
            {
                throw new ArgumentException(string.Format("Invalid key length. {0} bytes expected {1} provided.",
                    encDec.KeySize / 8, key.Length), "key");
            }

            if (iv.Length != ivSizeBytes)
            {
                throw new ArgumentException(string.Format("Invalid iv length. {0} bytes expected {1} provided.",
                    ivSizeBytes, iv.Length), "iv");
            }

            byte[] textbytes = ASCIIEncoding.UTF8.GetBytes(input);

            using (ICryptoTransform iEncProvider = encDec.CreateEncryptor(key, iv))
            {
                byte[] enc = iEncProvider.TransformFinalBlock(textbytes, 0, textbytes.Length);

                //TODO: am p2. For speed using System.Buffer.BlockCopy should be faster than LINQ.
                //Returning IV+encrypted_text. IV does not need to be encrypted.
                return Convert.ToBase64String(iv.Concat(enc).ToArray());
            }
        }

        private static string DecryptAes256(byte[] key, string input)
        {
            if (key.Length * 8 != encDec.KeySize)
            {
                throw new ArgumentException(string.Format("Invalid key length. {0} bytes expected {1} provided.",
                    encDec.KeySize / 8, key.Length), "key");
            }

            //First bytes are IV.
            byte[] encbytes = Convert.FromBase64String(input);

            //First bytes are IV.
            var iv = encbytes.Take(ivSizeBytes).ToArray();

            using (ICryptoTransform iDecProvider = encDec.CreateDecryptor(key, iv))
            {
                byte[] enc = iDecProvider.TransformFinalBlock(
                    encbytes, ivSizeBytes, encbytes.Length - ivSizeBytes);

                return ASCIIEncoding.UTF8.GetString(enc);
            }
        }

        private static string ByteToHex(IEnumerable<byte> bytes)
        {

            return string.Concat(bytes.Select(b => b.ToString("X2")));
        }

        private static byte[] HexToByte(string hex)
        {
            //TODO: am p2. Any cleaner way to convert? This is even more ugly.
            return Enumerable.Range(0, hex.Length)
                     .Where(x => x % 2 == 0)
                     .Select(x => Convert.ToByte(hex.Substring(x, 2), 16))
                     .ToArray();

        }
